main        proc near
var_10      = dword ptr -10h
            push ebp                      (1)
            mov ebp, esp
            and esp, 0FFFFFFF0h
            sub esp, 10h
            mov eax, offset aHelloWorld   (2)
            mov [esp+10h+var_10], eax
            call _printf
            mov eax, 0                    (3)
            leave                         (4)
            retn
main        endp

2 Preparing arguments and calling printf
3 Clean-up registers
